app.get('/', function (req, res) {
	var shader = req.query.shader;
	var ui = "/index.html";

	switch(shader) {
	case "cpu_meter":
		cpu_meter.start();
		break;
	case "notifier":
		var filename = path.join(__dirname + "/" + req.query.filename);
		notifier.start(filename, .3); // .3 secs
		break;
	case "aurora":
		aurora.start();
		break;
	case "unicorn":
		unicorn.start();
		break;
	case "fireplace":
		fireplace.start();
		break;
	case "ambilight":
		ambilight.start();
		break;
	case "timesquare":
		timesquare.start();
		break;
	case "stop":
		rgb_led_matrix.stop();
		break;
	case "start":
		rgb_led_matrix.start();
		break;
	case "clear":
		rgb_led_matrix.fadeOut();
		break;
	default:
		rgb_led_matrix.start();
	example = "default"
	}
	//res.sendfile(path.join(__dirname + "/" + ui));
})
